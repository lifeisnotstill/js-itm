 51 /**
 52  * Creates a new LatLng by converting coordinates from a source ellipsoid to a target one. <p>
 53  * This process involves:<p>
 54  *   1. converting the angular latlng to cartesian coordinates using latLngToPoint()<p>
 55  *   2. translating the XYZ coordinates using a translation, with special values supplied to this translation.<p>
 56  *   3. converting the translated XYZ coords back to a new angular LatLng<p>
 57  *<p>
 58  *  for example refer to {@link JSITM.itm2gps}
 59  *
 60  * @param {JSITM.Ellipsoid} from source elli
 61  * @param {JSITM.Ellipsoid} to
 62  * @param {JSITM.Translation} translation
 63  * @return {JSITM.LatLng}
 64  */
 65 JSITM.LatLng.prototype.convertGrid = function(from, to, translation){
 66 
 67     var point = from.latLngToPoint(this, 0);
 68     
 69     //removed 7 point Helmert translation (not needed in Israel's grids)
 70     var translated = translation.translate(point);
 71     
 72     return to.pointToLatLng(translated);
 73 }

 75 /**
 76  * Parses latitude and longtitude in a string into a new Latlng
 77  * @param {String} s
 78  * @return {JSITM.LatLng}
 79  */
 80 JSITM.latlngFromString = function(s) {
 81     var pattern = new RegExp("^(-?\\d+(?:\\.\\d*)?)(?:(?:\\s*[:,]?\\s*)|\\s+)(-?\\d+(?:\\.\\d*)?)$", "i")
 82     var latlng = s.match(pattern);
 83     if (latlng) {
 84         var lat = parseFloat(latlng[1], 10);
 85         var lng = parseFloat(latlng[2], 10);
 86         return new JSITM.LatLng(lat, lng);
 87     }
 88 
 89 	throw ("could not parse latlng")
 90 }

158 /**
159  * Creates a new Ellipsoid.<p>
160  * for more info see <a href="http://en.wikipedia.org/wiki/Reference_ellipsoid">http://en.wikipedia.org/wiki/Reference_ellipsoid</a>
161  *
162  * @constructor
163  * @param {Number} a length of the equatorial radius (the semi-major axis) in meters
164  * @param {Number} b length of the polar radius (the semi-minor axis) in meters
165  */
166 JSITM.Ellipsoid = function(a, b){
167     this.a = a;
168     this.b = b;
169     
170     // Compute eccentricity squared
171     this.e2 = (Math.pow(a, 2) - Math.pow(b, 2)) / Math.pow(a, 2);
172     
173 }

177 /**
178  * Creates a new LatLng containing an angular representation of a cartesian Point on the surface of the Ellipsoid.
179  *
180  * @param {JSITM.Point} point
181  * @return {JSITM.LatLng}
182  */
183 JSITM.Ellipsoid.prototype.pointToLatLng = function(point){
184 
185     var RootXYSqr = Math.sqrt(Math.pow(point.x, 2) + Math.pow(point.y, 2));
186     
187     var radlat1 = Math.atan2(point.z, (RootXYSqr * (1 - this.e2)));
188     
189     do {
190         var V = this.a / (Math.sqrt(1 - (this.e2 * Math.pow(Math.sin(radlat1), 2))));
191         var radlat2 = Math.atan2((point.z + (this.e2 * V * (Math.sin(radlat1)))), RootXYSqr);
192         if (Math.abs(radlat1 - radlat2) > 0.000000001) {
193             radlat1 = radlat2;
194         }
195         else {
196             break;
197         }
198     }
199     while (true);
200     
201     var lat = radlat2 * (180 / Math.PI);
202     var lng = Math.atan2(point.y, point.x) * (180 / Math.PI);
203     
204     return new JSITM.LatLng(lat, lng);
205 }

207 /**
208  * Creates a new Point object containing a cartesian representation of an angular LatLng on the surface of the Ellipsoid.
209  *
210  * @param {JSITM.LatLng} latlng
211  * @return {JSITM.Point}
212  */
213 JSITM.Ellipsoid.prototype.latLngToPoint = function(latlng){
214     // Convert angle measures to radians
215     var radlat = latlng.lat * (Math.PI / 180);
216     var radlng = latlng.lng * (Math.PI / 180);
217     
218     // Compute nu
219     var V = this.a / (Math.sqrt(1 - (this.e2 * (Math.pow(Math.sin(radlat), 2)))));
220     
221     // Compute XYZ
222     var x = (V + latlng.alt) * (Math.cos(radlat)) * (Math.cos(radlng));
223     var y = (V + latlng.alt) * (Math.cos(radlat)) * (Math.sin(radlng));
224     var z = ((V * (1 - this.e2)) + latlng.alt) * (Math.sin(radlat));
225     
226     return new JSITM.Point(x, y, z);
227 }

229 /**
230  * Cretaes a new TM (Transverse Mercator Projection).<br>
231  * For more info see <a href="http://en.wikipedia.org/wiki/Transverse_Mercator">http://en.wikipedia.org/wiki/Transverse_Mercator</a>
232  *
233  * @constructor
234  * @param {JSITM.Ellipsoid} reference ellipsoid
235  * @param {Number} e0 eastings of false origin in meters
236  * @param {Number} n0 northings of false origin in meters
237  * @param {Number} f0 central meridian scale factor
238  * @param {Number} lat0 latitude of false origin in decimal degrees.
239  * @param {Number} lng0 longitude of false origin in decimal degrees.
240  */
241 JSITM.TM = function(ellipsoid, e0, n0, f0, lat0, lng0){
242     //eastings (e0) and northings (n0) of false origin in meters; _
243     //central meridian scale factor (f0) and _
244     //latitude (lat0) and longitude (lng0) of false origin in decimal degrees.
245     
246     this.ellipsoid = ellipsoid;
247     this.e0 = e0;
248     this.n0 = n0;
249     this.f0 = f0;
250     this.lat0 = lat0;
251     this.lng0 = lng0;
252     
253     this.radlat0 = lat0 * (Math.PI / 180);
254     this.radlng0 = lng0 * (Math.PI / 180);
255     
256     this.af0 = ellipsoid.a * f0;
257     this.bf0 = ellipsoid.b * f0;
258     this.e2 = (Math.pow(this.af0, 2) - Math.pow(this.bf0, 2)) / Math.pow(this.af0, 2);
259     this.n = (this.af0 - this.bf0) / (this.af0 + this.bf0);
260     this.n2 = this.n * this.n; //for optimizing and clarity of Marc()
261     this.n3 = this.n2 * this.n; //for optimizing and clarity of Marc()
262     
263 }
264 
265 /**
266  * Compute meridional arc
267  * @private
268  * @param {Number} radlat latitude of meridian in radians
269  * @return {Number}
270  */
271 JSITM.TM.prototype.Marc = function(radlat){
272     
273     return (
274 		this.bf0 * (
275 			((1 + this.n + ((5 / 4) * this.n2) + ((5 / 4) * this.n3)) * (radlat - this.radlat0)) - 
276 			(((3 * this.n) + (3 * this.n2) + ((21 / 8) * this.n3)) * (Math.sin(radlat - this.radlat0)) * (Math.cos(radlat + this.radlat0))) + 
277 			((((15 / 8) * this.n2) + ((15 / 8) * this.n3)) * (Math.sin(2 * (radlat - this.radlat0))) * (Math.cos(2 * (radlat + this.radlat0)))) - 
278 			(((35 / 24) * this.n3) * (Math.sin(3 * (radlat - this.radlat0))) * (Math.cos(3 * (radlat + this.radlat0))))
279 		)
280 	);
281 		
282 
283 }

285 /**
286  * Returns the  initial value for Latitude in radians.
287  * @private
288  * @param {Number} y northings of point
289  * @return {Number}
290  */
291 JSITM.TM.prototype.InitialLat = function(y){
292   
293     var radlat1 = ((y - this.n0) / this.af0) + this.radlat0;
294 
295     var M = this.Marc(radlat1);
296     
297     var radlat2 = ((y - this.n0 - M) / this.af0) + radlat1;
298     
299     while (Math.abs(y - this.n0 - M) > 0.00001) {
300         radlat2 = ((y - this.n0 - M) / this.af0) + radlat1;
301         M = this.Marc(radlat2);
302         radlat1 = radlat2;
303     }
304     return radlat2;
305 }

307 /**
308  * Un-project Transverse Mercator eastings and northings back to latitude and longtitude.
309  * 
310  * @param {JSITM.Point} point
311  * @return {JSITM.LatLng} latitude and longtitude on the refernced ellipsoid's surface
312  */
313 
314 JSITM.TM.prototype.unproject = function(point){
315     //
316     //Input: - _
317     
318     //Compute Et
319     var Et = point.x - this.e0;
320     
321     //Compute initial value for latitude (PHI) in radians
322     var PHId = this.InitialLat(point.y);
323     
324     //Compute nu, rho and eta2 using value for PHId
325     var nu = this.af0 / (Math.sqrt(1 - (this.e2 * (Math.pow(Math.sin(PHId), 2)))));
326     var rho = (nu * (1 - this.e2)) / (1 - (this.e2 * Math.pow(Math.sin(PHId), 2)));
327     var eta2 = (nu / rho) - 1;
328     
329     //Compute Latitude
330     var VII = (Math.tan(PHId)) / (2 * rho * nu);
331     var VIII = ((Math.tan(PHId)) / (24 * rho * Math.pow(nu, 3))) * (5 + (3 * Math.pow(Math.tan(PHId), 2)) + eta2 - (9 * eta2 * (Math.pow(Math.tan(PHId), 2))));
332     var IX = (Math.tan(PHId) / (720 * rho * Math.pow(nu, 5))) * (61 + (90 * Math.pow(Math.tan(PHId), 2)) + (45 * (Math.pow(Math.tan(PHId), 4))));
333     
334     var lat = (180 / Math.PI) * (PHId - (Math.pow(Et, 2) * VII) + (Math.pow(Et, 4) * VIII) - (Math.pow(Et, 6) * IX));
335     
336     //Compute Longitude
337     var X = (Math.pow(Math.cos(PHId), -1)) / nu;
338     var XI = ((Math.pow(Math.cos(PHId), -1)) / (6 * Math.pow(nu, 3))) * ((nu / rho) + (2 * (Math.pow(Math.tan(PHId), 2))));
339     var XII = ((Math.pow(Math.cos(PHId), -1)) / (120 * Math.pow(nu, 5))) * (5 + (28 * (Math.pow(Math.tan(PHId), 2))) + (24 * (Math.pow(Math.tan(PHId), 4))));
340     var XIIA = ((Math.pow(Math.cos(PHId), -1)) / (5040 * Math.pow(nu, 7))) * (61 + (662 * (Math.pow(Math.tan(PHId), 2))) + (1320 * (Math.pow(Math.tan(PHId), 4))) + (720 * (Math.pow(Math.tan(PHId), 6))));
341     
342     var lng = (180 / Math.PI) * (this.radlng0 + (Et * X) - (Math.pow(Et, 3) * XI) + (Math.pow(Et, 5) * XII) - (Math.pow(Et, 7) * XIIA));
343     
344     var latlng = new JSITM.LatLng(lat, lng);
345     
346     return (latlng);
347 }

350 /**
351  * Project Latitude and longitude to Transverse Mercator coordinates
352  * @param {JSITM.LatLng} latitude and longtitude  to convert
353  * @return {JSITM.Point} projected coordinates 
354  */
355 JSITM.TM.prototype.project = function(latlng){
356     // Convert angle measures to radians
357     var RadPHI = latlng.lat * (Math.PI / 180);
358     var RadLAM = latlng.lng * (Math.PI / 180);
359     
360     var nu = this.af0 / (Math.sqrt(1 - (this.e2 * Math.pow(Math.sin(RadPHI), 2))));
361     var rho = (nu * (1 - this.e2)) / (1 - (this.e2 * Math.pow(Math.sin(RadPHI), 2)));
362     var eta2 = (nu / rho) - 1;
363     var p = RadLAM - this.radlng0;
364     var M = this.Marc(RadPHI);
365     
366     var I = M + this.n0;
367     var II = (nu / 2) * (Math.sin(RadPHI)) * (Math.cos(RadPHI));
368     var III = ((nu / 24) * (Math.sin(RadPHI)) * (Math.pow(Math.cos(RadPHI), 3))) * (5 - (Math.pow(Math.tan(RadPHI), 2)) + (9 * eta2));
369     var IIIA = ((nu / 720) * (Math.sin(RadPHI)) * (Math.pow(Math.cos(RadPHI), 5))) * (61 - (58 * (Math.pow(Math.tan(RadPHI), 2))) + (Math.pow(Math.tan(RadPHI), 4)));
370     
371     var y = I + (Math.pow(p, 2) * II) + (Math.pow(p, 4) * III) + (Math.pow(p, 6) * IIIA);
372     
373     var IV = nu * (Math.cos(RadPHI));
374     var V = (nu / 6) * (Math.pow(Math.cos(RadPHI), 3)) * ((nu / rho) - (Math.pow(Math.tan(RadPHI), 2)));
375     var VI = (nu / 120) * (Math.pow(Math.cos(RadPHI), 5)) * (5 - (18 * (Math.pow(Math.tan(RadPHI), 2))) + (Math.pow(Math.tan(RadPHI), 4)) + (14 * eta2) - (58 * (Math.pow(Math.tan(RadPHI), 2)) * eta2));
376     
377     var x = this.e0 + (p * IV) + (Math.pow(p, 3) * V) + (Math.pow(p, 5) * VI);
378     
379     return new JSITM.Point(x, y, 0)
380 }

421 /**
422  * Converts a point to an ITM reference.
423  * 
424  * @example 
425  *   JSITM.point2ItmRef(new JSITM.Point(200, 500)); // prints "200000500000"
426  *   JSITM.point2ItmRef(new JSITM.Point(200, 500), 3); // prints "200500"
427  * @param {JSITM.Point} point
428  * @param {Number} precision 3=km, 4=100 meter, 5=decameter 6=meter.  optional, default is 6,
429  * @return {String}
430  */
431 JSITM.point2ItmRef = function(point, precision){
432 
433 	function zeropad(num, len){
434 	    var str = new String(num);
435 	    while (str.length < len) {
436 	        str = '0' + str;
437 	    }
438 	    return str;
439 	}
440 	
441 	var p = precision || 6; 
442 
443     if (p < 3) 
444         p = 3;
445     if (p > 6) 
446         p = 6;
447     
448     var div = Math.pow(10, (6 - p));
449     var east = Math.round(point.x / div);
450     var north = Math.round(point.y / div);
451     return zeropad(east, p) + ' ' + zeropad(north, p);
452 	
453 }

455 /**
456  * Parses an ITM reference and returns a Point object.<p>
457  * throws an exception for invalid refernce!<p>
458  * <p>
459  * @example valid inputs:
460  *  200500
461  *  20005000
462  *  2000000500000
463  *  130:540
464  *  131550:44000
465  *  131 400
466  *  210222 432111
467  * 
468  * @param {String} s
469  * @return {JSITM.Point}
470  */
471 JSITM.itmRef2Point = function(s){
472     
473     var precision;
474     
475     for (precision = 6; precision >= 3; precision--) {
476         var pattern = new RegExp("^(\\d{" + precision + "})\\s*:?\\s*(\\d{" + precision + "})$", "i")
477         var ref = s.match(pattern);
478         if (ref) {
479             if (precision > 0) {
480                 var mult = Math.pow(10, 6 - precision);
481                 var x = parseInt(ref[1], 10) * mult;
482                 var y = parseInt(ref[2], 10) * mult;
483                 return new JSITM.Point(x, y);
484             }
485         }
486     }
487     
488 	throw "Could not parse reference";
489 }
